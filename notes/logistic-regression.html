3.3 判断模型好坏的方法

def getCrossEntropyValue(s, Y):
    arrayCrossEntropy = -1 * (np.dot(np.transpose(Y), np.log(s)) + np.dot((1-np.transpose(Y)), np.log(1-s))) / len(Y)
    return arrayCrossEntropy
